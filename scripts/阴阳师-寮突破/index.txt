# 检查再哪个页面
TAG START
SET VAR count 10
# 需要点击进攻页面
FIND_IMAGE image1.png image1 FIGHT CONTINU
# 成功页面
FIND_IMAGE image2.png image2 SUCCESS CONTINU
# 失败页面
FIND_IMAGE image4.png image4 FAILD CONTINU
# 等待开始
FIND_IMAGE image0.png image0 CHECK_TIMES SWIPE_CHECK

TAG CHECK_TIMES
# 看还有没有次数
FIND_IMAGE image3.png image3 HASNOT_TIMES CONTINU
FIND_IMAGE image0.png image0
CLICK image0
WAIT 2
FIND_IMAGE image1.png image1 CONTINU FLASH
TAG FIGHT
CLICK image1


TAG WAIT
WAIT 2
# 如果还在刚才页面
FIND_IMAGE image1.png image1 FLASH CONTINU 
FIND_IMAGE image4.png image4 FAILD CONTINU
FIND_IMAGE image2.png image2 SUCCESS WAIT

TAG FLASH
# 刷新寮突
LOG 刷新寮突
FIND_IMAGE image9.png image9
CLICK image9
WAIT 2
FIND_IMAGE image6.png image6
CLICK image6
WAIT 2
FIND_IMAGE image7.png image7
CLICK image7
WAIT 2
FIND_IMAGE image8.png image8
CLICK image8
WAIT 2
GO START

TAG SUCCESS
WAIT 3
CLICK image2
WAIT 3
GO START

TAG FAILD
WAIT 3
CLICK image4
WAIT 3
GO START

TAG SWIPE_CHECK
FIND_IMAGE image5.png image5 CONTINU PAGE_ERROR
CALC XY image5 y + 35 pos1
SWIP image5 pos1
WAIT 3
FIND_IMAGE image0.png image0 CHECK_TIMES CONTINU
CALC VAR count + 1 count
IF count > 20 HASNOT_ITEM SWIPE_CHECK


TAG HASNOT_ITEM

LOG 全部攻破
END

TAG HASNOT_TIMES
LOG 次数用完
LOG 等待1800秒(30分钟)
WAIT 1800
GO START

TAG PAGE_ERROR
ERROR 不在寮突页面
END
